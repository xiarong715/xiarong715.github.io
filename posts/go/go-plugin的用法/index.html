go plugin的用法

2020年12月20日
| Golang

1.编写plugin部分

创建文件testplugin.go

package main

import "fmt"

func init() {
	fmt.Println("load plugin")
}

// Hello hello
func Hello() {
	fmt.Println("Hello Plugin")
}

note: plugin的包名必须指定为main

2.编译plugin部分

1

go build --buildmode=plugin testplugin.go

编译后会生testplugin.so

3.编写主程序部分

package main

import "plugin"

func main() {
	p, err := plugin.Open("plugin/testplugin.so")
	if err != nil {
		panic(err)
	}
	f, err := p.Lookup("Hello")
	if err != nil {
		panic(err)
	}
	f.(func())()
}
